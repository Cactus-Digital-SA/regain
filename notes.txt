CREATE TABLE stopping_conditions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    entity_id INT,                      -- Refers to the entity (question, subscale, etc.)
    entity_type ENUM('question', 'subscale', 'test'),
    next_condition_id INT,              -- A condition ID that links to the next condition (for AND/OR logic chaining)
    operator ENUM('AND', 'OR'),         -- Logical operator for multiple conditions
    FOREIGN KEY (entity_id) REFERENCES entities(id)
);

CREATE TABLE condition_calculators (
    id INT PRIMARY KEY AUTO_INCREMENT,
    stopping_condition_id INT,                -- Refers to the stopping condition this calculator is part of
    condition_type ENUM('sum', 'comparison'), -- Type of calculation (sum, comparison)
    comparator ENUM('>=', '>', '=', '<', '<='), -- Comparison operator for numeric values
    subscale_id INT,                          -- For check conditions, stores subscale ID for comparison
    score_condition ENUM('>', '<', '>=', '<=', '='),  -- Score comparison condition for subscales
    FOREIGN KEY (stopping_condition_id) REFERENCES stopping_conditions(id),
    FOREIGN KEY (subscale_id) REFERENCES entities(id) -- subscale ID, for comparison
);

    question_ids TEXT,                        -- For sum conditions, stores a comma-separated list of question IDs
